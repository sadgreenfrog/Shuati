class Solution(object):
    def reverseWords(self, s):
#-----------两个指针进行去除空格----------------#
        s1 = list()
        for char in s:
            if char != ' ':
                s1.append(char)
            else: 
                continue
#-------------while进行string的reverse--------#
        i = 0
        j = len(s1) - 1
        while i < j:
            s1[i], s1[j] = s1[j], s1[i]
            i += 1
            j -= 1
